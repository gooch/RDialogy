    def self.run(options={}, std_err = false)
      cmd = dialog_string options
      if std_err
        begin
          tmp = Tempfile.new('tmp')
          cmd += ' 2> ' + tmp.path
        
          system cmd

          output = String.new
          while true
            output += tmp.readline
          end
        rescue EOFError
          if block_given?
            return yield output
          else
            return output
          end
        ensure
          tmp.close
        end
      else
        system cmd
      end
    end
